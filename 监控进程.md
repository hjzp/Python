import time
import datetime

def is_process_running(process_id):
        try:
                os.kill(process_id, 0)
                return True
        except OSError:
                return False

def is_process_alive(process_name):
        pid = os.system("pidof " + process_name)
        alive = is_process_running(pid)
        try:
                os.kill(pid, 0)
                return True
        except OSError:
                return False



while(1):
